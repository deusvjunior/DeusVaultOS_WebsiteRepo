const getSceneConfig = () => {
  if (subpageContext) {
    return { x: 15, y: 8, z: 20 }; // Zoomed detail view
  }

  switch (currentPage) {
    case "marketplace":
      return { x: 8, y: 2, z: 12 };
    case "features":
      return { x: -5, y: 3, z: 10 };
    default:
      return { x: 0, y: 0, z: 15 };
  }
};
